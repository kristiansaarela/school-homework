select
	linnaosa as linn,
	(
		( select sum(1laps) + sum(2last) * 2 + sum(3last) * 3 + sum(4last) * 4 from abielus where linnaosa = linn )
		+
		( select sum(1laps) + sum(2last) * 2 + sum(3last) * 3 + sum(4last) * 4 from teadmata where linnaosa = linn )
		+
		( select sum(1laps) + sum(2last) * 2 + sum(3last) * 3 + sum(4last) * 4 from lesk where linnaosa = linn )
		+
		select sum(1laps) + sum(2last) * 2 + sum(3last) * 3 + sum(4last) * 4
	) as kokku
from
	lahutatud
where
	linnaosa like '%linn%'
and
	linnaosa != 'Tudulinna vald'
group by
	linnaosa
order by
	kokku desc;
